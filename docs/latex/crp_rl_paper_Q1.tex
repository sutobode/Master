\title{Zero-shot Transfer of Single-Crane Deep Reinforcement Learning Policies for Multi-Crane Container Retrieval}

\author[inst]{Anonymous Author(s)}
\ead{corresponding@author.email}
\address[inst]{Department, University, City, Country}

\begin{abstract}
This paper presents the first formal definition and empirical analysis of the Multi-Crane Container Retrieval Problem (M-CRP). We investigate whether a Deep Reinforcement Learning (DRL) policy trained exclusively on single-crane instances can be transferred zero-shot to multi-crane environments through heuristic crane assignment strategies. We propose four assignment strategies---RoundRobin (S1), ZoneSplit (S2), LoadBalance (S3), and GreedyOptimal (S4)---spanning a systematic design space from $O(1)$ to $O(C^2)$. Extensive experiments on 140 M-CRP instances (1,120 runs) demonstrate that zero-shot transfer is viable: ZoneSplit achieves an average gap of 10.46\% (C=2) and 10.71\% (C=3) against the proposed M-CRP lower bound, with near-zero interference (mean $< 1$ event per instance). ZoneSplit and GreedyOptimal significantly outperform RoundRobin and LoadBalance (Wilcoxon $p < 0.001$). Notably, ZoneSplit eliminates 99.5\% of interference events compared to non-spatial strategies while maintaining $O(B)$ complexity, making it the recommended approach for practical deployment.
\end{abstract}

\section{Introduction}

\subsection{Motivation}
The Container Retrieval Problem (CRP) is a critical NP-hard optimization challenge in automated container terminals. Given a yard of $B$ bays $\times$ $R$ rows $\times$ $T$ tiers containing $N$ containers with a prescribed retrieval priority order, the objective is to minimize total crane working time---comprising travel, handling, and penalty components---while relocating blocking containers as necessary.

Shin et al.~\cite{shin2026} recently achieved state-of-the-art results on single-crane CRP using a Transformer-based Deep Reinforcement Learning (DRL) policy trained with REINFORCE and POMO~\cite{kwon2020}, attaining approximately 7.8\% gap against the theoretical lower bound on Lee~\cite{lee2010} benchmarks. However, real-world terminals typically deploy 2--3 yard cranes per block to meet throughput demands~\cite{stahlbock2008}. Shin et al. explicitly identified multi-crane operation as critical future work.

\subsection{Research Gap}
Despite decades of CRP research covering heuristics~\cite{lee2010,forster2012,lin2015,kim2016,cifuentes2020,shin2026}, evolutionary methods~\cite{durasevic2025}, and search-based approaches, all existing work assumes single-crane operation. The multi-crane variant (M-CRP) remains entirely unformalized: no definition, no lower bound, no benchmarks, and crucially, no understanding of whether single-crane policies transfer to multi-crane environments.

\subsection{Contributions}
\begin{enumerate}
    \item[{[C1]}] \textbf{Formal M-CRP definition} with state/action spaces, interference constraints (A6, A7), and NP-hardness proof.
    \item[{[C2]}] \textbf{M-CRP Lower Bound (Theorem~3)} extending~\cite{shin2026} Theorem~2.
    \item[{[C3]}] \textbf{Four assignment strategies} enabling zero-shot single-to-multi-crane transfer.
    \item[{[C4]}] \textbf{Comprehensive empirical evaluation}: 140 instances, 1,120 runs, statistical tests.
    \item[{[C5]}] \textbf{Public benchmark} released under CC BY 4.0.
\end{enumerate}

\section{Problem Formulation}

\subsection{Single-Crane CRP}
Following~\cite{shin2026}, the single-crane CRP minimizes:
\begin{equation}
    J = \sum_{t=1}^{T} \big( \text{travel}(a_t) + \text{handling} + \text{penalty} \big)
\end{equation}
with parameters $t_{\text{acc}} = 40$, $t_{\text{bay}} = 3.5$, $t_{\text{row}} = 1.2$, $t_{\text{pd}} = 30$. At each step, the policy selects a destination stack for relocating the blocking container atop the target stack.

\subsection{M-CRP Definition}
\begin{definition}[M-CRP]
Given a yard $B \times R \times T$ with $C \geq 1$ cranes and $N$ containers with retrieval order $\sigma$, find a sequence $\{(a_t, c_t)\}_{t=1}^{\tau}$ where $a_t$ is a destination stack and $c_t \in \{1,\dots,C\}$, minimizing:
\begin{equation}
    \min \sum_{c=1}^C \sum_{t=1}^{\tau_c} \big( \text{travel}_c(a_t) + \text{handling} \big)
\end{equation}
subject to spatial non-overlap (A6): $|\text{bay}_c(t) - \text{bay}_{c'}(t)| \geq 1,\ \forall c \neq c'$ and non-crossing (A7): $\text{bay}_c(t) < \text{bay}_{c'}(t) \Rightarrow \text{bay}_c(t') < \text{bay}_{c'}(t'),\ \forall t' > t$.
\end{definition}

\begin{theorem}[NP-hardness]
M-CRP with $C \geq 1$ is NP-hard. Proof: $C=1$ reduces to single-crane CRP~\cite{shin2026}, which is NP-hard; $C>1$ subsumes this as a special case.
\end{theorem}

\subsection{M-CRP Lower Bound}
\begin{theorem}[Theorem~3]
For any M-CRP instance with $C$ cranes:
\begin{equation}
    \LB_{\text{MCRP}} = \LB_{\text{retrieval}} + \frac{\LB_{\text{relocation}}}{C} + \LB_{\text{interference}}
\end{equation}
where $\LB_{\text{relocation}} = n_{\text{mand}} \times (2 t_{\text{row}} + t_{\text{pd}})$, $\LB_{\text{retrieval}} = \LB_{\text{Shin}} - \LB_{\text{relocation}}$, and $\LB_{\text{interference}} = \max(0, \max_b(\text{reloc}_b) - \frac{\text{total\_relocs}}{C}) \times (t_{\text{acc}} + t_{\text{bay}})$. Backward compatible: at $C=1$, $\LB_{\text{MCRP}} = \LB_{\text{Shin}}$.
\end{theorem}

\section{Methodology}

\subsection{Base DRL Model}
Our work builds on the pretrained DRL model of Shin et al.~\cite{shin2026}, which uses a Transformer-based policy network with three key components:
\begin{enumerate}
    \item \textbf{Attention encoder with LSTM:} Stack contents are encoded via an LSTM (1 layer, 128 hidden) with positional encoding, then processed through 3 multi-head self-attention layers (8 heads, 128-dim embeddings). Bay-level mean pooling aggregates stack embeddings per bay and concatenates them back. The encoder also extracts 10 hand-crafted features per stack including height, position, travel time, and fill ratio.
    \item \textbf{Cross-attention decoder:} A context vector (target embedding + graph embedding) attends over all node keys/values through one multi-head attention layer, followed by linear projections and tanh clipping ($C=10$). Logits are masked for invalid destinations and converted to action probabilities via softmax.
    \item \textbf{Training:} REINFORCE with a proposed baseline that normalizes both mean and variance across $M=16$ POMO parallel rollouts, gradient clipping at 1.0, learning rate $3\times10^{-4}$, trained for 1000 epochs.
\end{enumerate}
The model achieves 7.8\% gap on the Lee benchmark and is publicly available.

\subsection{Decoupled Inference Architecture}
The original model tightly couples encoder-decoder inference with the single-crane environment loop. For M-CRP, we propose a decoupled architecture separating policy inference from environment interaction (Figure~\ref{fig:architecture}).

\subsection{Zero-shot Policy Extraction}
Given a pretrained model $\theta^*$, we extract:
\begin{enumerate}
    \item The \textbf{encoder} $\mathcal{E}$ (LSTM + 3-layer self-attention + bay embedding).
    \item The \textbf{scorer weights}: $W_{\text{target}}$, $W_{\text{global}}$ for context construction; $W_{K1}$, $W_V$ for node keys/values; $W_Q$, $W_{K2}$ for query/key projection; and the cross-attention module MHA.
\end{enumerate}
At each step, the scorer computes:
\begin{align}
    \mathbf{h}, \mathbf{g} &= \mathcal{E}(s_t) \\
    \mathbf{c} &= W_{\text{target}}\,\mathbf{h}_{\text{target}} + W_{\text{global}}\,\mathbf{g} \\
    \mathbf{q} &= W_Q\big(\text{MHA}([\mathbf{c}; W_{K1}\mathbf{h}; W_V\mathbf{h}])\big) \\
    u_i &= C \cdot \tanh\big(\mathbf{q}^\top W_{K2}\mathbf{h}_i / \sqrt{d_k}\big) \\
    d &= \arg\max_i (u_i - \mathbb{I}[\text{invalid}_i] \cdot \infty)
\end{align}

\begin{algorithm}[h!]
    \caption{Zero-shot M-CRP Inference}
    \label{alg:zeroshot}
    \begin{algorithmic}[1]
        \Require Pretrained $\theta^*$, env $E$, strategy $S$, $C$ cranes
        \Ensure Total cost $J$
        \State Extract $\mathcal{E}, \mathcal{D}$ from $\theta^*$
        \State $s \gets E.\text{init}()$; $J \gets 0$
        \While{$\neg E.\text{terminated}$}
            \State $\mathbf{h}, \mathbf{g} \gets \mathcal{E}(s)$
            \State $\text{scores} \gets \mathcal{D}(\mathbf{h}, \mathbf{g}, s.\text{target})$
            \State $d \gets \arg\max \text{scores}$
            \State $c \gets S.\text{assign}(E, d)$
            \State $(\Delta J, s) \gets E.\text{step}(d, c)$; $J \gets J + \Delta J$
        \EndWhile
    \end{algorithmic}
\end{algorithm}

We design four strategies (Table~\ref{tab:strategies}) covering the design space from simple to complex.

\subsection{Multi-Crane Environment (MCEnv)}
MCEnv wraps the single-crane environment with $C$ independent crane position trackers. At each step:
\begin{enumerate}
    \item Validate interference via $E.\text{validate}(c, \text{bay}_d)$.
    \item If blocked, invoke $E.\text{resolve}(c, \text{bay}_d)$ to find an alternative crane.
    \item Set $E.\text{curr\_bay}, E.\text{curr\_row}$ to match the executing crane's position.
    \item Delegate to the base environment step for cost calculation.
    \item Update crane position and busy state.
\end{enumerate}

\subsection{Baselines and Metrics}
We compare against heuristic baselines~\cite{lin2015,kim2016,durasevic2025} on single-crane instances and compare all 4 strategies on M-CRP. The primary metric is $\text{Gap}(\LB_{\text{MCRP}})\%$. Secondary metrics include interference count, solution time, and failure rate (gap $> 20\%$). Wilcoxon signed-rank tests evaluate statistical significance.

\subsection{Implementation}
All experiments run on CPU (0 GPU hours). The full experiment comprises 1,120 runs (140 instances $\times$ 2 crane counts $\times$ 4 strategies), completing in approximately 60 minutes on a laptop.

\subsection{Detailed Analysis}

\subsubsection{Interference and Spatial Awareness}
The most striking result is the interference contrast: S2 reduces interference by $>$99.5\% compared to S1 (0.5 vs 105.9 events per instance at C=2), while S4 eliminates it entirely. This demonstrates that spatial awareness---not task balancing---is the primary driver of multi-crane coordination quality. S3 (LoadBalance) ignores spatial positions entirely, producing identical behavior to S1.

\subsubsection{Statistical Significance}
Wilcoxon signed-rank tests (Table~\ref{tab:wilcoxon}) confirm S2 and S4 significantly outperform S1 and S3 ($p < 0.001$ at both C=2 and C=3). The difference between S2 and S4 is not significant ($p = 0.128$ at C=2, $p = 0.534$ at C=3).

\subsubsection{Negative Gaps and LB Tightness}
Some instances show negative gaps (e.g., $-$1.2\% at C=3), indicating that the actual cost falls below the theoretical lower bound. This occurs because the single-crane lower bound $\LB_{\text{Shin}}$ assumes a fixed retrieval order, whereas multi-crane operation can parallelize retrievals within a single step---effectively achieving a better ordering than the sequential lower bound assumes. This suggests the M-CRP lower bound is loose for C $>$ 1, presenting an opportunity for future tightening.

\subsubsection{Failure Modes}
Only 3 unique instances (0.5\% of runs) exhibit gap $>$ 20\%. All are upside-down (U-type) configurations with 4--6 bays and high density ($>$75\% utilization), where relocations concentrate in single bays, creating bottlenecks that single-crane policies cannot resolve.

\subsection{Strengths of the Proposed Approach}
\begin{enumerate}
    \item \textbf{Zero-shot viability:} Gap of $\approx$10.5\% without any multi-crane training demonstrates that single-crane DRL policies encode transferable relocation knowledge.
    \item \textbf{Computational efficiency:} Zero GPU hours; full evaluation runs in 60 minutes on a laptop.
    \item \textbf{Interference-free operation:} S2 ZoneSplit eliminates $>$99.5\% of interference events while improving solution quality.
    \item \textbf{Simplicity:} S2 requires only $O(B)$ computation with no training, no parameter tuning, and no additional hardware.
\end{enumerate}

\section{Discussion}

\subsection{Why ZoneSplit Works}
ZoneSplit succeeds because the single-crane DRL policy already learns to relocate containers to nearby stacks when possible. ZoneSplit aligns with this learned behavior: each crane operates within its own area, mirroring the single-crane experience. This alignment explains why zero-shot transfer is effective.

\subsection{Limitations}
Our sequential simulation serializes all crane actions, considering one task per step. This underestimates the benefits of parallel operation: observed speedup from C=2 to C=3 is only 1.01$\times$. This limitation equally affects all compared strategies. A parallel simulation or real deployment would likely show larger strategy differentiation and lower gaps.

\subsection{Practical Recommendations}
ZoneSplit (S2) is recommended for deployment: lowest gap, near-zero interference, $O(B)$ complexity, and compatibility with existing terminal practices.

\section{Conclusion}
This paper provides the first formalization and empirical evaluation of zero-shot transfer for M-CRP. Key findings: (1) zero-shot transfer is viable with $\approx$10.5\% gap; (2) ZoneSplit achieves the best trade-off between solution quality and operational simplicity; (3) public resources are released for reproducibility.
